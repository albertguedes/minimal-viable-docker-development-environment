# Crontab for Minimal Viable Docker Development Environment
#
# Installation:
#   crontab crontab.txt
#   Or: cat crontab.txt | crontab -
#
# Logs:
#   /var/log/backup.log - Backup execution log
#   /var/log/uptime.log - Uptime check failures

# Backup database daily at 2 AM
0 2 * * * /home/albert/Development/DOCKER/minimal_viable/backup/backup.sh >> /var/log/backup.log 2>&1

# Check uptime every 5 minutes
*/5 * * * * curl -sf http://localhost:8080/health > /dev/null || echo "DOWN: $(date)" >> /var/log/uptime.log

# Note: Edit the absolute path above to match your installation directory
